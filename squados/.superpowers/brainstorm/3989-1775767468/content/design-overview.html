1 · Banco de dados

-- Nova tabela de junção
CREATE TABLE user_sectors (
  user_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  sector_id UUID REFERENCES sectors(id) ON DELETE CASCADE,
  assigned_at TIMESTAMPTZ DEFAULT now(),
  PRIMARY KEY (user_id, sector_id)
);
-- Nova coluna em profiles
ALTER TABLE profiles
  ADD COLUMN active_sector_id UUID REFERENCES sectors(id);
-- Migrar sector_id existente para user_sectors
INSERT INTO user_sectors (user_id, sector_id)
  SELECT id, sector_id FROM profiles WHERE sector_id IS NOT NULL;

• profiles.sector_id mantido para compatibilidade com código existente → será deprecado
• active_sector_id é o que o agente de IA lê para saber o contexto atual do usuário
• RLS: somente admin/master_admin pode inserir/deletar em user_sectors
